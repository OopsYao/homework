% Reformat title, author and date
\LetLtxMacro{\OldTitle}{\title}

    \IfNoValueTF{#1}{%
        \OldTitle[#2]{\textnormal{\textbf{#2}}}
    }{%
        \OldTitle[#1]{\textnormal{\textbf{#2}}}
    }
}

\LetLtxMacro{\OldAuthor}{\author}
\RenewDocumentCommand{\author}{ o m }{%
    \IfNoValueTF{#1}{%
        \OldAuthor[#2]{\small{#2}}
    }{%
        \OldAuthor[#1]{\small{#2}}
    }
}
